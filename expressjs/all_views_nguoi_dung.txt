function mapNguoiDung(user) {
  return {
    taiKhoan: user.tai_khoan,
    matKhau: user.mat_khau,
    email: user.email,
    soDt: user.so_dt,
    maNhom: user.ma_nhom,

    hoTen: user.ho_ten,
  };
}

/** JWT subject — luôn có giá trị (tai_khoan hoặc email). */
function tokenSubject(user) {
  return user.tai_khoan || user.email;
}

/** Hỗ trợ bcrypt ($2...) hoặc mật khẩu plain (DB mẫu users). */
function verifyPassword(plain, stored) {
  if (!stored) return false;
  if (stored.startsWith("$2a$") || stored.startsWith("$2b$") || stored.startsWith("$2y$")) {
    return bcrypt.compareSync(plain, stored);
  }
  return plain === stored;
}

function defaultTaiKhoan(email, hoTen) {
  if (email?.includes("@")) return email.split("@")[0];
  return hoTen?.replace(/\s+/g, "").toLowerCase().slice(0, 50) || "user";
}

async function findUserByTaiKhoanOrEmail(taiKhoan) {
  if (!taiKhoan?.trim()) return null;
  const key = taiKhoan.trim();
  return prisma.nguoi_dung.findFirst({
    where: {
      OR: [{ tai_khoan: key }, { email: key }],
    },
  });
}

- **Khắc phục lỗi Giao Dịch rỗng**: Hàm `luuGiaoDich` giờ đây đã được gọi cho **TẤT CẢ** các phương thức thanh toán (cả Thẻ và Mã QR). Do code cũ đã bỏ sót việc gọi API này khi thanh toán bằng Thẻ, nên bảng `giao_dich` trong Database bị trống. Lỗi này đã được triệt để loại bỏ!
- **Khắc phục kẹt Step 2 (QR)**: Trước đây, sau khi thanh toán QR thành công, hệ thống không chịu chuyển sang màn hình Success (Step 3). Bây giờ, hệ thống sẽ tự động `dispatch({ type: "NEXT_STEP" })` để đưa khách hàng tới màn hình chúc mừng!
- **Khắc phục lỗi Book Another**: Chỉnh sửa thứ tự hoạt động của nút "Book Another" (`await navigate` trước khi `RESET`). Tránh trường hợp màn hình chớp báo lỗi "Phiên đã kết thúc" khiến người dùng hiểu nhầm là lỗi hệ thống.
- **Đảm bảo Vé Của Tôi (My Tickets) hiển thị**: Do trước đây người dùng gặp lỗi kẹt không lưu được vé mới, phần "Vé của tôi" luôn báo rỗng. Sau bản fix n
  - **Thêm/Cập nhật Người Dùng**: Form hiện pop-up cho phép thiết lập loại người dùng (Khách Hàng / Quản Trị Viên) với các thông tin đầy đủ.
- Tích hợp các API gọi trực tiếp qua `quanLyNguoiDungApi` ở `cinema-api.ts`.

### [Client] Vé Của Tôi (My Tickets)
- **Sửa lỗi hiển thị**: `TicketsModal` hiện đã nhóm tất cả vé mua cùng lúc thành 1 hàng duy nhất thay vì hiển thị rải rác từng ghế.

- Thay đổi hành vi nút "ĐẶT VÉ KHÁC" ở màn hình thanh toán thành công.
- Thay vì chỉ khởi tạo lại state làm kẹt màn hình báo "Phiên đã kết thúc", hệ thống giờ đã tự động xóa State và **điều hướng trực tiếp (navigate) về màn hình chính "/"** cực kỳ mượt mà.

## 2. Backend Security & Logic (Node.js/Express)

### Bảo Mật Endpoint
- **Phân quyền API**: Đã áp dụng Middleware `restrictTo("QuanTri")` cho toàn bộ các Router nhạy cảm ở cả Phim và Người Dùng. 
  - Chỉ có Admin (Quản Trị) mới có quyền tạo lịch chiếu, xóa phim, sửa phim, và quản lý người dùng.
- Middleware sẽ chặn đứng bất kỳ truy cập trái phép nào không có Token Admin hợp lệ.

### Gom Nhóm Vé Phía Backend
- API `ThongTinTaiKhoan` trong `quan-ly-nguoi-dung.service.js` đã được tinh chỉnh để gom nhóm các bản ghi đặt ghế theo **Mã Lịch Chiếu** và **Thời Gian Đặt**. Trả về một Object duy nhất chứa mảng danh sách ghế `danhSachGhe` cho Frontend render.
